groups = df2.groupby(['ID/Name'])
industries = df1['Vertical'].unique()
cmap = {industries[i]:i / len(industries) for i in range(len(industries))}
names = []  # Node labels
mrr = []   # Size of each node
term = []  # Average term
contract_cnt = []
for k, v in groups:
    mrr.append(v['Total MRR'].sum() / 100)
    term.append(v['Term'].mean())
    names.append(k)
    contract_cnt.append(v['Service Count'].iloc[0])
colors = [cmap[v] for v in [df1[df1['ID/Name'] == label]['Vertical'].iloc[0] for label in names]]
colorguide = sorted(list(set(zip(colors, [df1[df1['ID/Name'] == label]['Vertical'].iloc[0] for label in names]))),
                    key=lambda t: t[0])
plt.figure(figsize=(16, 2))
plt.scatter(np.arange(len(colorguide)), np.ones(len(colorguide)),
            alpha=0.4, s=1000, c=[c[0] for c in colorguide], cmap=plt.get_cmap('Dark2'))
for i in range(len(colorguide)):
    plt.annotate(colorguide[i][1], xy=(i, 1),
                 xytext=(i, 30 + (-1)**i * 10), textcoords='offset points',
                ha='right', va='bottom',
                arrowprops = dict(arrowstyle='->', connectionstyle = 'arc3,rad=0'))
plt.xlim(-1, 20)
plt.show()
plt.figure(figsize=(32, 12))
plt.scatter(contract_cnt, term, s=mrr, alpha=0.4, c=colors, cmap=plt.get_cmap('Dark2'))
for label, x, y in zip(names, contract_cnt, term):
    plt.annotate(label, xy=(x, y))
plt.xlabel('Contract Count')
plt.ylabel('Average Contract Length')
plt.show()
